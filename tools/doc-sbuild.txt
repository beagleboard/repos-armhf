#debian bullseye:
sudo sbuild-shell source:bullseye-armhf-sbuild

echo "deb http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian bullseye main" > /etc/apt/sources.list ; \
echo "deb-src http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian bullseye main" >> /etc/apt/sources.list ; \
echo "deb [trusted=yes] http://repos.rcn-ee.net/debian bullseye main" >> /etc/apt/sources.list ; \
echo "deb-src [trusted=yes] http://repos.rcn-ee.net/debian bullseye main" >> /etc/apt/sources.list ; \
cat /etc/apt/sources.list ; \
apt update ; apt install -yq bbb.io-keyring ; \
echo "deb http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian bullseye main" > /etc/apt/sources.list ; \
echo "deb-src http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian bullseye main" >> /etc/apt/sources.list ; \
echo "deb [arch=armhf signed-by=/usr/share/keyrings/rcn-ee-archive-keyring.gpg] http://repos.rcn-ee.net/debian bullseye main" >> /etc/apt/sources.list ; \
echo "deb-src [arch=armhf signed-by=/usr/share/keyrings/rcn-ee-archive-keyring.gpg] http://repos.rcn-ee.net/debian bullseye main" >> /etc/apt/sources.list ; \
cat /etc/apt/sources.list ; \
apt update ; apt-get clean ; echo "done..." ; exit
