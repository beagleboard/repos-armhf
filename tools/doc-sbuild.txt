#debian jessie:
sudo sbuild-shell source:jessie-armhf-sbuild

echo "deb http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian jessie main" > /etc/apt/sources.list ; \
echo "deb-src http://proxy.gfnd.rcn-ee.org:3142/deb.debian.org/debian jessie main" >> /etc/apt/sources.list ; \
echo "deb http://repos.rcn-ee.com/debian jessie main" >> /etc/apt/sources.list ; \
echo "deb-src http://repos.rcn-ee.com/debian jessie main" >> /etc/apt/sources.list ; \
cat /etc/apt/sources.list ; \
apt-get update ; apt-get --yes --allow-unauthenticated install apt-transport-https rcn-ee-archive-keyring ; \
apt-get update ; apt-get clean ; echo "done..." ; exit
